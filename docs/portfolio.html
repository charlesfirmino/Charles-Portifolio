Exemplo de consulta SQL:

SELECT
    Cliente.nome,
    SUM(Pagamento.valorTotal) AS valor_devido
FROM
    Cliente
JOIN
    Locacao ON Cliente.idCliente = Locacao.idCliente
JOIN
    Pagamento ON Locacao.idPagamento = Pagamento.idPagamento
WHERE
    Pagamento.estado = 'Pendente'
GROUP BY
    Cliente.idCliente, Cliente.nome
ORDER BY
    Cliente.nome ASC;
